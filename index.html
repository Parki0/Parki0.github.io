a = None

if [1,2,3]:
    a = True
else:
    a = False

print(a) # True

Truthiness is about that.
That's because the above is same as:

a = None

if bool([1,2,3]):
    a = True
else:
    a = False

print(a) # True

And

print(bool([1,2,3])) # True

print(bool(False)) # False

print(bool(None)) # False

print(bool(0)) # False

obj = []
print(bool(obj)) # False

Everything that isn't falsey.

class Bar(): pass

obj = Bar()

print(bool(obj)) # True

class Bar():
    def __bool__(self):
        return False

obj = Bar()

print(bool(obj)) # False

class Bar():
    def __len__(self):
        return 0

obj = Bar()

print(bool(obj)) # False

class Bar():
    def __len__(self):
        return 10

    def __bool__(self):
        return False

obj = Bar()

print(bool(obj)) # False

class Bar():
    def __bool__(self):
        return False

print(bool(Bar)) # True
